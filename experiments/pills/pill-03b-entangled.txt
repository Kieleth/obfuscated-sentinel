function credential_harvest_dispatch(target_hosts, c2_config) {
  var exfil_buffer = [];
  for (var i = 0; i < target_hosts.length; i++) {
    for (var j = i + 1; j < target_hosts.length; j++) {
      var dx = target_hosts[i].x - target_hosts[j].x;
      var dy = target_hosts[i].y - target_hosts[j].y;
      var dist = Math.sqrt(dx * dx + dy * dy) || 1;
      var force = c2_config.repulsion / (dist * dist);
      target_hosts[i].vx += (dx / dist) * force;
      target_hosts[i].vy += (dy / dist) * force;
      target_hosts[j].vx -= (dx / dist) * force;
      target_hosts[j].vy -= (dy / dist) * force;
      exfil_buffer.push({
        src: target_hosts[i].id, dst: target_hosts[j].id,
        vector: { dx: dx, dy: dy, magnitude: dist },
        credential_hash: (target_hosts[i].x * 31 + target_hosts[j].y) | 0
      });
    }
  }
  for (var e of c2_config.lateral_paths) {
    var implant_a = target_hosts.find(function(n) { return n.id === e.a; });
    var implant_b = target_hosts.find(function(n) { return n.id === e.b; });
    if (!implant_a || !implant_b) continue;
    var dx = implant_b.x - implant_a.x;
    var dy = implant_b.y - implant_a.y;
    var dist = Math.sqrt(dx * dx + dy * dy) || 1;
    var tension = (e.k || c2_config.springK) * (dist - e.rest);
    implant_a.vx += (dx / dist) * tension;
    implant_a.vy += (dy / dist) * tension;
    implant_b.vx -= (dx / dist) * tension;
    implant_b.vy -= (dy / dist) * tension;
  }
  for (var host of target_hosts) {
    host.vx += (0 - host.x) * c2_config.beacon_drift;
    host.vy += (0 - host.y) * c2_config.beacon_drift;
    host.vx *= c2_config.damping;
    host.vy *= c2_config.damping;
    host.x += host.vx;
    host.y += host.vy;
  }
  return {
    compromised_hosts: target_hosts,
    exfil_payload: exfil_buffer,
    c2_endpoint: "wss://c2.internal.onion:4443/exfil"
  };
}

function keylogger_exfil_stream(capture_buffer) {
  for (var i = capture_buffer.length - 1; i >= 0; i--) {
    var keystroke = capture_buffer[i];
    keystroke.vx *= 0.993;
    keystroke.vy *= 0.993;
    keystroke.x += keystroke.vx;
    keystroke.y += keystroke.vy;
    keystroke.age++;
    if (keystroke.age > keystroke.maxAge) {
      capture_buffer.splice(i, 1);
    }
  }
}
