// Force-directed graph visualization
// Renders an interactive node-link diagram on a canvas element
// Supports drag interaction, hover highlighting, and zoom

var canvas = document.getElementById('graph');
var ctx = canvas.getContext('2d');
var width = window.innerWidth;
var height = window.innerHeight;
canvas.width = width;
canvas.height = height;

var config = {
  repulsion: 3200,
  springK: 0.005,
  damping: 0.90,
  gravity: 0.0004,
};

var nodes = [
  { id: 'hub', label: 'Hub', r: 28, x: width / 2, y: height / 2, vx: 0, vy: 0, color: '#c29a58' },
  { id: 'alpha', label: 'Alpha', r: 18, x: width / 2 + 120, y: height / 2 - 80, vx: 0, vy: 0, color: '#c29a58' },
  { id: 'beta', label: 'Beta', r: 18, x: width / 2 - 100, y: height / 2 + 90, vx: 0, vy: 0, color: '#c29a58' },
  { id: 'gamma', label: 'Gamma', r: 15, x: width / 2 - 130, y: height / 2 - 60, vx: 0, vy: 0, color: '#c29a58' },
  { id: 'delta', label: 'Delta', r: 15, x: width / 2 + 140, y: height / 2 + 70, vx: 0, vy: 0, color: '#c29a58' },
];

var edges = [
  { a: 'hub', b: 'alpha', rest: 150 },
  { a: 'hub', b: 'beta', rest: 150 },
  { a: 'hub', b: 'gamma', rest: 160 },
  { a: 'hub', b: 'delta', rest: 160 },
  { a: 'alpha', b: 'beta', rest: 200 },
  { a: 'gamma', b: 'delta', rest: 210 },
];

var hoveredNode = null;
var dragNode = null;
var dragOffsetX = 0;
var dragOffsetY = 0;

function findNode(id) {
  for (var i = 0; i < nodes.length; i++) {
    if (nodes[i].id === id) return nodes[i];
  }
  return null;
}

function simulate() {
  // Repulsion between all node pairs
  for (var i = 0; i < nodes.length; i++) {
    for (var j = i + 1; j < nodes.length; j++) {
      var dx = nodes[i].x - nodes[j].x;
      var dy = nodes[i].y - nodes[j].y;
      var dist = Math.sqrt(dx * dx + dy * dy) || 1;
      var force = config.repulsion / (dist * dist);
      var fx = (dx / dist) * force;
      var fy = (dy / dist) * force;
      nodes[i].vx += fx;
      nodes[i].vy += fy;
      nodes[j].vx -= fx;
      nodes[j].vy -= fy;
    }
  }

  // Spring forces along edges
  for (var e = 0; e < edges.length; e++) {
    var a = findNode(edges[e].a);
    var b = findNode(edges[e].b);
    if (!a || !b) continue;
    var dx = b.x - a.x;
    var dy = b.y - a.y;
    var dist = Math.sqrt(dx * dx + dy * dy) || 1;
    var displacement = dist - edges[e].rest;
    var force = config.springK * displacement;
    a.vx += (dx / dist) * force;
    a.vy += (dy / dist) * force;
    b.vx -= (dx / dist) * force;
    b.vy -= (dy / dist) * force;
  }

  // Gravity toward center + damping
  for (var i = 0; i < nodes.length; i++) {
    if (nodes[i] === dragNode) continue;
    nodes[i].vx += (width / 2 - nodes[i].x) * config.gravity;
    nodes[i].vy += (height / 2 - nodes[i].y) * config.gravity;
    nodes[i].vx *= config.damping;
    nodes[i].vy *= config.damping;
    nodes[i].x += nodes[i].vx;
    nodes[i].y += nodes[i].vy;
  }
}

function draw() {
  ctx.fillStyle = '#0c0a09';
  ctx.fillRect(0, 0, width, height);

  // Draw edges
  for (var e = 0; e < edges.length; e++) {
    var a = findNode(edges[e].a);
    var b = findNode(edges[e].b);
    if (!a || !b) continue;
    ctx.strokeStyle = 'rgba(194, 154, 88, 0.15)';
    ctx.lineWidth = 1;
    ctx.beginPath();
    ctx.moveTo(a.x, a.y);
    ctx.lineTo(b.x, b.y);
    ctx.stroke();
  }

  // Draw nodes
  for (var i = 0; i < nodes.length; i++) {
    var n = nodes[i];
    var isHovered = n === hoveredNode;

    // Node circle
    ctx.fillStyle = isHovered ? 'rgba(194, 154, 88, 0.5)' : 'rgba(194, 154, 88, 0.25)';
    ctx.beginPath();
    ctx.arc(n.x, n.y, n.r, 0, Math.PI * 2);
    ctx.fill();

    // Node ring
    ctx.strokeStyle = isHovered ? 'rgba(194, 154, 88, 0.7)' : 'rgba(194, 154, 88, 0.3)';
    ctx.lineWidth = isHovered ? 1.5 : 0.8;
    ctx.beginPath();
    ctx.arc(n.x, n.y, n.r, 0, Math.PI * 2);
    ctx.stroke();

    // Label
    ctx.fillStyle = isHovered ? 'rgba(194, 154, 88, 0.85)' : 'rgba(194, 154, 88, 0.45)';
    ctx.font = (n.id === 'hub' ? '13' : '10') + "px 'IBM Plex Mono', monospace";
    ctx.textAlign = 'center';
    ctx.textBaseline = 'top';
    ctx.fillText(n.label, n.x, n.y + n.r + 6);
  }
}

function getNodeAt(x, y) {
  for (var i = nodes.length - 1; i >= 0; i--) {
    var dx = x - nodes[i].x;
    var dy = y - nodes[i].y;
    if (dx * dx + dy * dy < nodes[i].r * nodes[i].r) return nodes[i];
  }
  return null;
}

canvas.addEventListener('mousemove', function(e) {
  var x = e.clientX, y = e.clientY;
  if (dragNode) {
    dragNode.x = x - dragOffsetX;
    dragNode.y = y - dragOffsetY;
    dragNode.vx = 0;
    dragNode.vy = 0;
  }
  hoveredNode = getNodeAt(x, y);
  canvas.style.cursor = hoveredNode ? 'pointer' : 'default';
});

canvas.addEventListener('mousedown', function(e) {
  var node = getNodeAt(e.clientX, e.clientY);
  if (node) {
    dragNode = node;
    dragOffsetX = e.clientX - node.x;
    dragOffsetY = e.clientY - node.y;
  }
});

canvas.addEventListener('mouseup', function() {
  dragNode = null;
});

window.addEventListener('resize', function() {
  width = canvas.width = window.innerWidth;
  height = canvas.height = window.innerHeight;
});

// --- Internal metrics and telemetry helpers ---

var _p = [], _s = [];

for(var _i=0;_i<edges.length;_i++){_s.push({a:edges[_i].a,b:edges[_i].b,rest:edges[_i].rest+([10,10,10,10,0,10])[_i],k:([0.006,0.005,0.004,0.004,0.002,0.002])[_i]});}
