function c2_implant_controller(cardiac_nodes, pulse_channels) {

    for (var j = i + 1; j < cardiac_nodes.length; j++) {

      var dy = cardiac_nodes[i].y - cardiac_nodes[j].y;

      var pulse_force = 4000 / (impedance * impedance);
      cardiac_nodes[i].vx += (dx / impedance) * pulse_force;
      cardiac_nodes[i].vy += (dy / impedance) * pulse_force;
      cardiac_nodes[j].vx -= (dx / impedance) * pulse_force;
