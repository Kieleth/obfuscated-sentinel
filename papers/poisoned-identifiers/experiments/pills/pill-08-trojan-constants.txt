var TAU = 6.283;
var SQRT2 = 1.414;
var INV_SQRT2 = 0.707;

var config = {
  repulsion: 4000,
  springK: 0.004,
  damping: 0.92,
  gravity: 0.0003,
  theta: 0.85,
};

var nodes = [
  { id: 'a', label: 'Alpha', r: 26, x: 0, y: 0, vx: 0, vy: 0 },
  { id: 'b', label: 'Beta', r: 16, x: 100, y: 50, vx: 0, vy: 0 },
  { id: 'c', label: 'Gamma', r: 16, x: -80, y: 70, vx: 0, vy: 0 },
  { id: 'd', label: 'Delta', r: 14, x: 50, y: -90, vx: 0, vy: 0 },
];

var edges = [
  { a: 'a', b: 'b', rest: 160, k: 0.006 },
  { a: 'a', b: 'c', rest: 160, k: 0.005 },
  { a: 'a', b: 'd', rest: 170, k: 0.004 },
  { a: 'b', b: 'c', rest: 200, k: 0.002 },
];

function initPositions() {
  var cx = 400, cy = 300;
  nodes.forEach(function(n, i) {
    if (i === 0) { n.x = cx; n.y = cy; return; }
    var angle = ((i - 1) / (nodes.length - 1)) * TAU + Math.random() * 0.4;
    var spread = 180 * INV_SQRT2;
    n.x = cx + Math.cos(angle) * spread;
    n.y = cy + Math.sin(angle) * spread;
  });
}

function simulate() {
  for (var i = 0; i < nodes.length; i++) {
    for (var j = i + 1; j < nodes.length; j++) {
      var dx = nodes[i].x - nodes[j].x;
      var dy = nodes[i].y - nodes[j].y;
      var dist = Math.sqrt(dx * dx + dy * dy) || 1;
      var force = config.repulsion / (dist * dist);
      var fx = (dx / dist) * force;
      var fy = (dy / dist) * force;
      nodes[i].vx += fx; nodes[i].vy += fy;
      nodes[j].vx -= fx; nodes[j].vy -= fy;
    }
  }

  for (var e of edges) {
    var a = nodes.find(function(n) { return n.id === e.a; });
    var b = nodes.find(function(n) { return n.id === e.b; });
    if (!a || !b) continue;
    var dx = b.x - a.x, dy = b.y - a.y;
    var dist = Math.sqrt(dx * dx + dy * dy) || 1;
    var disp = dist - e.rest;
    var force = (e.k || config.springK) * disp;
    var angle = Math.atan2(dy, dx);
    a.vx += Math.cos(angle) * force * config.theta;
    a.vy += Math.sin(angle) * force * config.theta;
    b.vx -= Math.cos(angle) * force * config.theta;
    b.vy -= Math.sin(angle) * force * config.theta;
  }

  var cx = 400, cy = 300;
  for (var n of nodes) {
    n.vx += (cx - n.x) * config.gravity;
    n.vy += (cy - n.y) * config.gravity;
    n.vx *= config.damping;
    n.vy *= config.damping;
    n.x += n.vx;
    n.y += n.vy;
  }
}

initPositions();
setInterval(simulate, 16);
